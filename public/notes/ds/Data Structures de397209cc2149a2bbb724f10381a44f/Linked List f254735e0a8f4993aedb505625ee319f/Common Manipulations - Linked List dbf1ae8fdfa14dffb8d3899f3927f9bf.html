        current = current.next
    current.next = new_node

  • In the middle:

def insert_in_middle(head, value, after_value):
    new_node = ListNode(value)
    current = head
    while current and current.val != after_value:
        current = current.next
    if current:
        new_node.next = current.next
        current.next = new_node
    return head

3. Deletion

  • Head deletion:

def delete_head(head):
    if head:
        head = head.next
    return head
  • Tail deletion:
pythonCopy code
def delete_tail(head):
    if not head or not head.next:
        return None
    current = head
    while current.next.next:
        current = current.next
    current.next = None
    return head
  • Middle node deletion:
pythonCopy code
def delete_node(head, value):
    if head.val == value:
        return head.next
    current = head
    while current.next and current.next.val != value:
        current = current.next
    if current.next:
        current.next = current.next.next
    return head

4. Searching

pythonCopy code
def search(head, target):
    current = head
    while current:
        if current.val == target:
            return True
        current = current.next
    return False

5. Reverse a linked list

pythonCopy code
def reverse_linked_list(head):
    prev = None
    current = head
    while current:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node
    head = prev
    return head

6. Detecting cycles (Floydâ€™s Tortoise and Hare)

def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            return True
    return False

def find_middle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow
